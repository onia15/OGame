cmake_minimum_required(VERSION 3.16)

project(OniaGame)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

option(PRODUCTION_BUILD "Make this a production build" OFF)

# SIMD flags
if(MSVC)
    add_compile_options(/arch:AVX2)
else()
    add_compile_options(-mavx2 -mfma)
endif()

# MSVC runtime selection
if(MSVC)
    set(CMAKE_MSVC_RUNTIME_LIBRARY
        "$<$<CONFIG:Debug>:MultiThreadedDebug>$<$<CONFIG:Release>:MultiThreaded>"
    )
endif()

# Collect sources
file(GLOB_RECURSE MY_SOURCES CONFIGURE_DEPENDS
    "${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp"
)

add_executable(${CMAKE_PROJECT_NAME} ${MY_SOURCES})

# Resource paths
if(PRODUCTION_BUILD)
    target_compile_definitions(${CMAKE_PROJECT_NAME} PUBLIC
        RESOURCES_PATH="./resources/"
        PRODUCTION_BUILD=1
    )
else()
    target_compile_definitions(${CMAKE_PROJECT_NAME} PUBLIC
        RESOURCES_PATH="${CMAKE_CURRENT_SOURCE_DIR}/resources/"
        PRODUCTION_BUILD=0
    )
endif()

# Platform-specific options
if(MSVC)
    target_compile_options(${CMAKE_PROJECT_NAME} PUBLIC /D_CRT_SECURE_NO_WARNINGS)

    if(PRODUCTION_BUILD)
        set_target_properties(${CMAKE_PROJECT_NAME} PROPERTIES
            LINK_FLAGS "/SUBSYSTEM:WINDOWS /ENTRY:mainCRTStartup"
        )
    endif()

elseif(MINGW)
    if(PRODUCTION_BUILD)
        target_link_options(${CMAKE_PROJECT_NAME} PRIVATE -mwindows)
    endif()

elseif(APPLE OR LINUX)
    if(PRODUCTION_BUILD)
        target_link_options(${CMAKE_PROJECT_NAME} PRIVATE -s)
    endif()
endif()

target_include_directories(${CMAKE_PROJECT_NAME} PUBLIC
    "${CMAKE_CURRENT_SOURCE_DIR}/src/"
)
